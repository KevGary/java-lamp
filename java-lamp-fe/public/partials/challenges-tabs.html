Greatest Product


Implement an algorithm to determine the greatest product between two elements in an array.

Test Code

describe('Greatest Product', function(){
  it("returns greates product of pairs in array", function (){
    expect(code.greatestProductOfPairs([2,3,4,-5,92,100])).toEqual(9200);
    expect(code.greatestProductOfPairs([-2,-3,-4,-5,-92,100])).toEqual(460);
  });
});

Answer 1

function(array){
  var max = 0;
  for(var i = 0; i < array.length; i++){
    for(var j = i+1; j < array.length; j++){
      var curr = array[i] * array[j];
      if(curr > max){
        max = curr;
      }
    }
  }
  return max;
}
